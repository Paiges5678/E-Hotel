CREATE TABLE HotelChain(
ChainID SERIAL PRIMARY KEY,
ChainName VARCHAR,
ChainAddress VARCHAR,
NumHotels INTEGER,
ChainContactEmail VARCHAR,
ChainContactPhone VARCHAR
);

CREATE TABLE Hotel(
HotelID SERIAL PRIMARY KEY,
ChainID INTEGER REFERENCES HotelChain(ChainID) ON DELETE CASCADE,
StarRating INTEGER,
NumRooms INTEGER,
HotelAddress VARCHAR,
HotelContactEmail VARCHAR,
HotelContactPhone VARCHAR
);

CREATE TABLE Room(
RoomID SERIAL PRIMARY KEY,
HotelID INTEGER REFERENCES Hotel(HotelID) ON DELETE CASCADE,
Price INTEGER CHECK (price >= 0),
Amenities VARCHAR,
Capacity INTEGER,
View VARCHAR,
ExtendableStatus BOOLEAN,
ProblemDamageStatus VARCHAR
);

CREATE TABLE Inst_Amenities(
RoomID PRIMARY KEY REFERENCES Room(RoomID) ON DELETE CASCADE
);

CREATE TABLE Pays_For(
PayID SERIAL PRIMARY KEY,
RentingID INTEGER REFERENCES Renting(RentingID) ON DELETE CASCADE,
CustomerID INTEGER REFERENCES Customer(CustomerID) ON DELETE CASCADE,
CardNumber INTEGER
);

CREATE TABLE Customer(
CustomerID SERIAL PRIMARY KEY,
Name_First VARCHAR,
Name_Last VARCHAR,
Name_Middle VARCHAR,
CustAddress VARCHAR,
IDType VARCHAR,
IDNumber INTEGER,
CustRegistrationDate VARCHAR
);

CREATE TABLE Employee(
EmployeeID SERIAL PRIMARY KEY,
HotelID INTEGER REFERENCES Hotel(HotelID) ON DELETE CASCADE,
Name_First VARCHAR,
Name_Last VARCHAR,
Name_Middle VARCHAR,
EmpAddress VARCHAR,
Position VARCHAR,
IsManager BOOLEAN
);

CREATE TABLE Booking(
BookID SERIAL PRIMARY KEY,
CustomerID INTEGER REFERENCES Customer(CustomerID) ON DELETE CASCADE,
HotelID INTEGER REFERENCES Hotel(HotelID) ON DELETE CASCADE,
RoomID INTEGER REFERENCES Room(RoomID) ON DELETE CASCADE,
BookingDate VARCHAR,
StartDate VARCHAR,
EndDate VARCHAR
);

CREATE TABLE Renting(
RentingID SERIAL PRIMARY KEY,
CustomerID INTEGER REFERENCES Customer(CustomerID) ON DELETE CASCADE,
BookID INTEGER REFERENCES Booking(BookID) ON DELETE CASCADE,
HotelID INTEGER REFERENCES Hotel(HotelID) ON DELETE CASCADE,
RoomID INTEGER REFERENCES Room(RoomID) ON DELETE CASCADE,
EmployeeID INTEGER REFERENCES Employee(EmployeeID) ON DELETE CASCADE,
CheckinDate VARCHAR,
CheckoutDate VARCHAR
);

CREATE TABLE Archive(
ArchiveID SERIAL PRIMARY KEY,
CopyRentingID INTEGER,
CopyBookID INTEGER,
CopyCustomerName_First VARCHAR,
CopyCustomerName_Middle VARCHAR,
CopyCustomerName_Last VARCHAR,
CopyCustomerID INTEGER,
CopyHotelID INTEGER,
CopyRoomID INTEGEr,
CopyBookingDate VARCHAR,
CopyCheckinDate VARCHAR,
CopyCheckoutDate VARCHAR
);
