--Trigger 1 - archive booking after its tranformed into a rentin
CREATE OR REPLACE FUNCTION archive_booking()
RETURNS TRIGGER AS $$
BEGIN
    INSERT INTO Archive(ArchiveID, CopyRentingID, CopyBookID, CopyCustomerID, CopyRoomID, CopyBookingDate)
    VALUES (nextval('archive_seq'), NEW.RentingID, NEW.BookID, New.CustomerID, NEW.HotelID, NEW.RoomID,
        (SELECT BookingDate FROM Booking WHERE BookID = NEW.BookID));
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

--Trigger 2 - don't allow booking if StartDate is after EndDate
CREATE OR REPLACE FUNCTION check_date()
RETURNS TRIGGER AS $$
BEGIN
    IF NEW.StartDate >=New.EndDate THEN
        RAISE EXCEPTION 'End date must be after Start date.';
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trigger_check_date
BEFORE INSERT OR UPDATE ON Booking
FOR EACH ROW EXECUTE FUNCTION check_date();

--Trigger 3 - each hotel has only 1 manager
CREATE OF REPLACE FUNCTION single_manager()
RETURNS TRIGGER AS $$
BEGIN
    IF NEW.IsManager = TRUE THEN
        UPDATE Employee
        SET IsManager = FALSE
        WHERE HotelID = NEW.HotelID
        AND IsManager = TRUE
        AND EmployeeID != NEW.EmployeeID;
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;


CREATE TRIGGGER trigger_single_manager
BEFORE INSERT OR UPDATE ON Employee
FOR EACH ROW EXECUTE FUNCTION single_manager()
