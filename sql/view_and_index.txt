--INDEX

-- Index for fast availability search (hotel, room, date range)
CREATE INDEX idx_booking_dates ON Booking(HotelID, RoomNumber, StartDate, EndDate);

-- Index for customer lookups by SIN (common in employee interface)
CREATE INDEX idx_customer_sin ON Customer(sin_number);

-- Index for reporting on check-in dates
CREATE INDEX idx_renting_checkin ON Renting(CheckinDate);

CREATE INDEX IF NOT EXISTS ixRoomPrice ON Room(Price);


--VIEWS
-- View 1: Available rooms per area (area is the city from HotelAddress and it's always the last word after the last comma)

CREATE VIEW AvailableRoomsByArea AS
SELECT 
    SUBSTRING(h.HotelAddress FROM ',\s*([^,]+)$') AS area,
    r.HotelID,
    r.RoomNumber,
    r.Capacity,
    r.Price
FROM Room r
JOIN Hotel h ON r.HotelID = h.HotelID
WHERE r.ProblemDamageStatus = FALSE
  AND NOT EXISTS (
      SELECT 1 FROM Booking b
      WHERE b.HotelID = r.HotelID
        AND b.RoomNumber = r.RoomNumber
        AND b.StartDate <= CURRENT_DATE
        AND b.EndDate >= CURRENT_DATE
  );

-- View 2: Aggregated capacity of all rooms per hotel
CREATE VIEW HotelTotalCapacity AS
SELECT 
    HotelID,
    SUM(Capacity) AS TotalCapacity
FROM Room
GROUP BY HotelID;
